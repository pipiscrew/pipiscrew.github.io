CREATE TABLE members (
  member_id int(11) NOT NULL AUTO_INCREMENT,
  email varchar(50) COLLATE utf8_unicode_ci DEFAULT NULL,
  password varchar(60) COLLATE utf8_unicode_ci DEFAULT NULL,
